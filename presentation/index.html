  body:      String,
  updatedAt: { 'type': Date, 'default': Date.now },
  createdAt: { 'type': Date, 'default': Date.now }
});

mongoose.model('Article', schema, 'article');

module.exports = function (req, res) {
  var model = new Article({
    title: req.body.title,
    body: req.body.body,
    url: req.body.url
  });

  model.save(function (err, data) {
    if (err) {
      return res.status(500).send(err);
    }

    res.status(201).send(data);
  });
};

Asynchrone Handlung (ohne Promises)

Article.find({ id: 1 }, function (err, res) {
  if (err) {
    return handleError(err);
  }

  handleResult(res);
});

Asynchrone Handlung (mit Promises)

Article.find({ id: 1 })
  .then(handleResult)
  .catch(handleError)
;

Fortlaufende Aktionen (ohne Promises)

function loadArticleWithAuthor(callback) {
  Article.all(function (err, result) {
    if (err) {
      return handleError(err);
    }

    var count = result.length,
        i, article;

    for (i = 0; i < count; i++) {
      article = result[i];

      User.find({ id: article.authorId }, function (err, user) {
        if (userErr) {
          handleError(userErr);
        } else {
          article.author = user;
        }

        count--;

        if (0 === count) {
          callback(result);
        }
      });
    }
  });
}

function loadArticleWithAuthor() {
  return Article
    .all()
    .then(function (res) {
      return Promise.all(res.map(function (article) {
        return User.find({ id: article.authorId })
          .then(function (user) {
            article.author = user;
          })
        ;
      }));
    })
    .catch(handleError)
  ;
}

Controller

function ArticleCreateController($scope) {
  $scope.flashMessage = null;
  $scope.title        = null;
  $scope.body         = null;

  $scope.save = function save() {
    api()
      .method('POST').resource('article')
      .data('title', $scope.title).data('body', $scope.body)
      .request()
      .then(function () {
        $scope.title        = null;
        $scope.body         = null;
        $scope.flashMessage = { type: 'success', content: 'Erfolg!' };
      })
    ;
  };
}
